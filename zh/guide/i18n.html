    # 多语言支持

    # 主题多语言配置

    可以通过 themeConfig.locales 来配置. 该选项接受同样的 { path: config } 格式的值.

    module.exports = {
  locales: { /* ... */ },
  themeConfig: {
    locales: {
      '/': {
        tagsLabel: 'Tags',
        readMore: 'Read More...',
        pageTags: 'Tags',
        formatDateLabel: {
          yearAgo: ' year ago',
          monthAgo: ' month ago',
          weekAgo: ' week ago',
          dayAgo: ' day ago',
          hourAgo: ' hour ago',
          minuteAgo: ' minute ago',
          secondAgo: ' second ago',
        }
      },
      '/zh/': {
        tagsLabel: '标签云',
        readMore: '继续阅读 »',
        pageTags: '标签',
        formatDateLabel: {
          yearAgo: '年前',
          monthAgo: '月前',
          weekAgo: '周前',
          dayAgo: '日前',
          hourAgo: '小时前',
          minuteAgo: '分钟前',
          secondAgo: '刚刚',
        }
      }
    }
  }
}
